#1. Contare quanti iscritti ci sono stati ogni anno

SELECT YEAR(`enrolment_date`) AS YEARS , COUNT(`id`) AS COUNTER
FROM `students`
GROUP BY YEAR(`enrolment_date`)
 ;


#2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio

SELECT  `office_number` AS  UFFICIO ,COUNT(`id`) AS  INSEGNANTI_COUNTER 
FROM `teachers`
GROUP BY `office_number`;

#3. Calcolare la media dei voti di ogni appello d'esame

SELECT `exam_id` AS APELLO_DI_ESAME ,  AVG(`vote`) AS AVG
 FROM `exam_student`
GROUP BY `exam_id`;

#4. Contare quanti corsi di laurea ci sono per ogni dipartiment 

SELECT `department_id` AS  DEPARTIMENT ,  COUNT(`id`) AS `CORSI_DI_LAUREA`
 FROM `degrees`
 GROUP BY `department_id`
 ;
